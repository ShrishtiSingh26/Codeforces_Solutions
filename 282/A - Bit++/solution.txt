import sys
 
def main():
    # Read all inputs from standard input
    input = sys.stdin.read
    data = input().split()
    
    if not data:
        return
 
    n = int(data[0])
    x = 0
    
    # Process each statement
    for i in range(1, n + 1):
        statement = data[i]
        if '+' in statement:
            x += 1
        else:
            x -= 1
            
    print(x)
 
if __name__ == '__main__':
    main()
